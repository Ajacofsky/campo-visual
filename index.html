import streamlit as st import cv2 import numpy as np from streamlit_image_coordinates import streamlit_image_coordinates st.set_page_config(layout="wide", page_title="Peritaje Pro - Calibraci√≥n Manual") st.title("‚öñÔ∏è Analizador de Incapacidad - Calibraci√≥n por Referencia") img_file = st.sidebar.file_uploader("1. Subir Campo Visual", type=['jpg', 'png', 'jpeg']) if img_file is not None: file_bytes = np.asarray(bytearray(img_file.read()), dtype=np.uint8) img_orig = cv2.imdecode(file_bytes, 1) h, w = img_orig.shape[:2] # --- INTERFAZ DE CALIBRACI√ìN --- st.sidebar.markdown("### üõ†Ô∏è Pasos de Calibraci√≥n") # Usamos session_state para guardar los clics if "puntos" not in st.session_state: st.session_state.puntos = [] # Lista para guardar [Centro, Marca 60¬∞] if len(st.session_state.puntos) < 2: instruccion = "Hac√© clic en el CENTRO (Cruce X, Y)" if not st.session_state.puntos else "Ahora hac√© clic en la marca de 60¬∞ (Derecha)" st.warning(f"üëâ {instruccion}") else: if st.sidebar.button("üîÑ Reiniciar Calibraci√≥n"): st.session_state.puntos = [] st.rerun() # Captura de clics sobre la imagen coords = streamlit_image_coordinates(img_orig, key="calibrador") if coords and len(st.session_state.puntos) < 2: nuevo_punto = (coords["x"], coords["y"]) if nuevo_punto not in st.session_state.puntos: st.session_state.puntos.append(nuevo_punto) st.rerun() # --- PROCESAMIENTO SI TENEMOS LOS PUNTOS --- if len(st.session_state.puntos) == 2: cx, cy = st.session_state.puntos[0] # Centro x60, y60 = st.session_state.puntos[1] # Punto 60 grados # Calcular Escala (P√≠xeles por grado) distancia_px = np.sqrt((x60 - cx)**2 + (y60 - cy)**2) px_por_grado = distancia_px / 60 # Calcular Rotaci√≥n (por si el papel est√° torcido) angulo_base = np.degrees(np.arctan2(y60 - cy, x60 - cx)) # Procesamiento de imagen para detectar cuadrados umbral = st.sidebar.slider("Sensibilidad de Negros", 50, 255, 145) gris = cv2.cvtColor(img_orig, cv2.COLOR_BGR2GRAY) _, binaria = cv2.threshold(gris, umbral, 255, cv2.THRESH_BINARY_INV) kernel = np.ones((3,3), np.uint8) binaria = cv2.morphologyEx(binaria, cv2.MORPH_OPEN, kernel) img_final = img_orig.copy() puntos_totales = 0 # Generar los 5 anillos (10, 20, 30, 40, 50 grados) for i in range(5): r_ext = (i + 1) * 10 * px_por_grado r_int = i * 10 * px_por_grado # 8 sectores de 45¬∞ cada uno for s in range(8): a_inicio = s * 45 + angulo_base - 0 # Ajuste de rotaci√≥n mask = np.zeros((h, w), dtype=np.uint8) cv2.ellipse(mask, (cx, cy), (int(r_ext), int(r_ext)), 0, a_inicio, a_inicio + 45, 255, -1) cv2.circle(mask, (cx, cy), int(r_int), 0, -1) # Evaluaci√≥n de contenido area_negra = np.count_nonzero(cv2.bitwise_and(binaria, mask)) area_total = np.count_nonzero(mask) if area_total > 0: ocupacion = (area_negra / area_total) * 100 color = None pts = 0 # TU REGLA: Amarillo > 70% | Celeste < 70% if ocupacion >= 10: # Ajustado: 10% de masa suele ser >70% de cuadraditos color = (0, 255, 255) # Amarillo pts = 10 elif 1.5 < ocupacion < 10: color = (255, 255, 0) # Celeste pts = 5 if color: puntos_totales += pts overlay = img_final.copy() cv2.ellipse(overlay, (cx, cy), (int(r_ext), int(r_ext)), 0, a_inicio, a_inicio + 45, color, -1) cv2.circle(overlay, (cx, cy), int(r_int), (255,255,255), -1) img_final = cv2.addWeighted(overlay, 0.4, img_final, 0.6, 0) # Dibujar l√≠neas de referencia de calibraci√≥n cv2.line(img_final, (cx, cy), (x60, y60), (255, 0, 0), 2) cv2.drawMarker(img_final, (cx, cy), (255, 0, 0), cv2.MARKER_CROSS, 20, 2) st.success(f"‚úÖ Calibraci√≥n lograda: {round(px_por_grado, 2)} px/grado") st.markdown(f"## üìä SUMA: {puntos_totales} pts | Incapacidad: {round((puntos_totales/320)*100, 1)}%") st.image(cv2.cvtColor(img_final, cv2.COLOR_BGR2RGB), use_container_width=True) else: # Mostrar imagen original para clics si no est√° calibrado st.image(img_orig, caption="Esperando calibraci√≥n...", use_container_width=True) else: st.info("Sub√≠ el informe para iniciar la calibraci√≥n manual.")
