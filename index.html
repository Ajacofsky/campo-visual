import streamlit as st import cv2 import numpy as np st.set_page_config(layout="wide", page_title="Peritaje Pro V26") st.title("丘뒲잺 Calibraci칩n de M치xima Visibilidad") img_file = st.sidebar.file_uploader("Subir Informe", type=['jpg', 'png', 'jpeg']) if img_file: file_bytes = np.asarray(bytearray(img_file.read()), dtype=np.uint8) img_orig = cv2.imdecode(file_bytes, 1) h, w = img_orig.shape[:2] # --- CONTROLES LATERALES --- st.sidebar.header("游꿢 1. Centro") cx = st.sidebar.slider("Centro X", 0, w, w // 2) cy = st.sidebar.slider("Centro Y", 0, h, h // 2) st.sidebar.header("游늺 2. Escala (Equidistante)") # El usuario marca el 60. El sistema calcula el paso de 10. x60 = st.sidebar.slider("Alinear con Marca 60춿", cx, w, cx + 250) # MATEM츼TICA PURA: Paso exacto cada 10 grados px_cada_10 = (x60 - cx) / 6.0 st.sidebar.header("游댧 3. Detecci칩n") umbral = st.sidebar.slider("Sensibilidad Negros", 50, 255, 140) # --- RENDERIZADO --- img_res = img_orig.copy() gris = cv2.cvtColor(img_orig, cv2.COLOR_BGR2GRAY) _, binaria = cv2.threshold(gris, umbral, 255, cv2.THRESH_BINARY_INV) puntos_totales = 0 # 1. ANALIZAR Y PINTAR SECTORES (10춿 a 40춿) for i in range(1, 5): r_ext = int(i * px_cada_10) r_int = int((i - 1) * px_cada_10) for s in range(8): ang = s * 45 mask = np.zeros((h, w), dtype=np.uint8) cv2.ellipse(mask, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, 255, -1) cv2.circle(mask, (cx, cy), r_int, 0, -1) area_negra = np.count_nonzero(cv2.bitwise_and(binaria, mask)) area_total = np.count_nonzero(mask) if area_total > 0: ocu = (area_negra / area_total) * 100 if ocu >= 8: # Amarillo overlay = img_res.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, (0, 255, 255), -1) img_res = cv2.addWeighted(overlay, 0.4, img_res, 0.6, 0) puntos_totales += 10 elif 1.2 < ocu < 8: # Celeste overlay = img_res.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, (255, 255, 0), -1) img_res = cv2.addWeighted(overlay, 0.4, img_res, 0.6, 0) puntos_totales += 5 # 2. DIBUJAR GRILLA NEGRA REFORZADA (L칤neas de 10춿 a 60춿) # Dibujamos de mayor a menor para asegurar que los centros no se borren for i in range(6, 0, -1): radio = int(i * px_cada_10) # EFECTO CONTRASTE: Borde blanco de 8px y centro negro de 4px # Esto garantiza que se vea en zonas oscuras y claras cv2.circle(img_res, (cx, cy), radio, (255, 255, 255), 6) # Borde blanco cv2.circle(img_res, (cx, cy), radio, (0, 0, 0), 3) # L칤nea negra interna # 3. EJES CARTESIANOS Y MARCAS # Eje X e Y con el mismo efecto de alto contraste cv2.line(img_res, (cx, 0), (cx, h), (255, 255, 255), 4) cv2.line(img_res, (cx, 0), (cx, h), (0, 0, 0), 2) cv2.line(img_res, (0, cy), (w, cy), (255, 255, 255), 4) cv2.line(img_res, (0, cy), (w, cy), (0, 0, 0), 2) # Marca de validaci칩n en 60 grados (Flecha Roja) cv2.drawMarker(img_res, (x60, cy), (0, 0, 255), cv2.MARKER_TRIANGLE_DOWN, 30, 5) # --- MOSTRAR RESULTADOS --- st.subheader(f"游늵 Resultado: {puntos_totales} pts | Incapacidad: {round((puntos_totales/320)*100, 1)}%") st.image(cv2.cvtColor(img_res, cv2.COLOR_BGR2RGB), use_container_width=True) st.write(f"游늺 **INFO:** Cada marca de 10춿 est치 a {round(px_cada_10, 1)} p칤xeles del centro.") else: st.info("Sub칤 la imagen para aplicar la grilla de alta visibilidad.")
