import streamlit as st import cv2 import numpy as np st.set_page_config(layout="wide", page_title="Peritaje Pro V19") st.title("‚öñÔ∏è Analizador de Incapacidad - Calibraci√≥n Manual") # --- BARRA LATERAL --- img_file = st.sidebar.file_uploader("1. Subir Campo Visual", type=['jpg', 'png', 'jpeg']) if img_file: # Cargar imagen file_bytes = np.asarray(bytearray(img_file.read()), dtype=np.uint8) img_orig = cv2.imdecode(file_bytes, 1) h, w = img_orig.shape[:2] st.sidebar.markdown("---") st.sidebar.subheader("üéØ Ajuste del Centro") # Sliders para mover la cruz roja al centro (0,0) del informe cx = st.sidebar.slider("Centro Horizontal (X)", 0, w, w // 2) cy = st.sidebar.slider("Centro Vertical (Y)", 0, h, h // 2) st.sidebar.markdown("---") st.sidebar.subheader("üìè Ajuste de Escala") # Slider para ajustar el tama√±o de los anillos seg√∫n las marcas de 60¬∞ px_cada_10 = st.sidebar.slider("Tama√±o de Grilla (Pixeles/10¬∞)", 10, 150, 50) umbral = st.sidebar.slider("Sensibilidad de Negros", 50, 255, 145) # --- PROCESAMIENTO --- gris = cv2.cvtColor(img_orig, cv2.COLOR_BGR2GRAY) _, binaria = cv2.threshold(gris, umbral, 255, cv2.THRESH_BINARY_INV) img_res = img_orig.copy() puntos_totales = 0 # Crear los 4 anillos (10¬∞, 20¬∞, 30¬∞, 40¬∞) como en tu documento for i in range(1, 5): r_ext = i * px_cada_10 r_int = (i - 1) * px_cada_10 # Analizar 8 sectores por anillo for s in range(8): ang = s * 45 mask = np.zeros((h, w), dtype=np.uint8) cv2.ellipse(mask, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, 255, -1) cv2.circle(mask, (cx, cy), r_int, 0, -1) # Conteo de masa s√≥lida (cuadrados negros) sector_negro = cv2.bitwise_and(binaria, mask) area_negra = np.count_nonzero(sector_negro) area_total = np.count_nonzero(mask) if area_total > 0: ocupacion = (area_negra / area_total) * 100 color, pts = None, 0 # Regla de puntos if ocupacion >= 8: # Cuadrado s√≥lido detectado color, pts = (0, 255, 255), 10 # Amarillo elif 1.5 < ocupacion < 8: color, pts = (255, 255, 0), 5 # Celeste if color: puntos_totales += pts overlay = img_res.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, color, -1) cv2.circle(overlay, (cx, cy), r_int, (255,255,255), -1) img_res = cv2.addWeighted(overlay, 0.4, img_res, 0.6, 0) # Dibujar borde del anillo cv2.circle(img_res, (cx, cy), r_ext, (200, 200, 200), 1) # Dibujar Cruz de Gu√≠a cv2.drawMarker(img_res, (cx, cy), (0, 0, 255), cv2.MARKER_CROSS, 40, 2) # --- MOSTRAR --- st.subheader(f"üìä Suma Total: {puntos_totales} pts") st.image(cv2.cvtColor(img_res, cv2.COLOR_BGR2RGB), use_container_width=True) st.markdown(""" **Instrucciones de uso:** 1. Mov√©s los sliders de **X e Y** hasta que la cruz roja est√© en el centro (0,0). 2. Mov√©s el slider de **Tama√±o de Grilla** hasta que los anillos coincidan con las marcas de tu informe. 3. El c√°lculo de puntos se actualiza solo. """) else: st.info("Sub√≠ la imagen en el panel izquierdo para comenzar.")
