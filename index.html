import streamlit as st import cv2 import numpy as np st.set_page_config(layout="wide", page_title="Peritaje Pro V20") st.title("丘뒲잺 Analizador de Incapacidad - Visualizaci칩n Mejorada") img_file = st.sidebar.file_uploader("1. Subir Campo Visual", type=['jpg', 'png', 'jpeg']) if img_file: file_bytes = np.asarray(bytearray(img_file.read()), dtype=np.uint8) img_orig = cv2.imdecode(file_bytes, 1) h, w = img_orig.shape[:2] # --- CONTROLES --- st.sidebar.subheader("游꿢 Posicionamiento") cx = st.sidebar.slider("Centro X", 0, w, w // 2) cy = st.sidebar.slider("Centro Y", 0, h, h // 2) px_cada_10 = st.sidebar.slider("Tama침o Grilla (Pix/10춿)", 10, 150, 55) st.sidebar.subheader("游댧 Detecci칩n") umbral = st.sidebar.slider("Sensibilidad de Negros", 50, 255, 145) opacidad = st.sidebar.slider("Opacidad de Colores", 0.1, 1.0, 0.5) # --- PROCESAMIENTO --- gris = cv2.cvtColor(img_orig, cv2.COLOR_BGR2GRAY) _, binaria = cv2.threshold(gris, umbral, 255, cv2.THRESH_BINARY_INV) img_res = img_orig.copy() puntos_totales = 0 # Dibujamos de afuera hacia adentro para que el centro quede "arriba" for i in range(1, 5): r_ext = i * px_cada_10 r_int = (i - 1) * px_cada_10 for s in range(8): ang = s * 45 mask = np.zeros((h, w), dtype=np.uint8) cv2.ellipse(mask, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, 255, -1) cv2.circle(mask, (cx, cy), int(r_int), 0, -1) # Conteo de masa s칩lida area_negra = np.count_nonzero(cv2.bitwise_and(binaria, mask)) area_total = np.count_nonzero(mask) if area_total > 0: ocupacion = (area_negra / area_total) * 100 color, pts = None, 0 if ocupacion >= 8: # Amarillo (10 pts) color, pts = (0, 255, 255), 10 elif 1.2 < ocupacion < 8: # Celeste (5 pts) color, pts = (255, 255, 0), 5 if color: puntos_totales += pts overlay = img_res.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, color, -1) cv2.circle(overlay, (cx, cy), int(r_int), (255,255,255), -1) img_res = cv2.addWeighted(overlay, opacidad, img_res, 1 - opacidad, 0) # --- DIBUJAR BORDES NEGROS FUERTES --- # L칤neas radiales (divisiones de 45춿) x_line = int(cx + r_ext * np.cos(np.radians(ang))) y_line = int(cy + r_ext * np.sin(np.radians(ang))) cv2.line(img_res, (cx, cy), (x_line, y_line), (0, 0, 0), 2) # L칤nea negra gruesa # C칤rculo del anillo en negro fuerte cv2.circle(img_res, (cx, cy), int(r_ext), (0, 0, 0), 2) # Marca del Centro (0,0) reforzada cv2.drawMarker(img_res, (cx, cy), (0, 0, 255), cv2.MARKER_CROSS, 50, 3) # --- RESULTADOS --- st.subheader(f"游늵 Suma: {puntos_totales} pts | Incapacidad: {round((puntos_totales/320)*100, 1)}%") st.image(cv2.cvtColor(img_res, cv2.COLOR_BGR2RGB), use_container_width=True) else: st.info("Sub칤 la imagen para ver la grilla reforzada.")
