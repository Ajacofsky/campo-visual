import streamlit as st import cv2 import numpy as np st.set_page_config(layout="wide", page_title="Peritaje Pro V29") st.title("âš–ï¸ CalibraciÃ³n Equidistante - Grilla Roja") img_file = st.sidebar.file_uploader("Subir Informe", type=['jpg', 'png', 'jpeg']) if img_file: # Carga de imagen file_bytes = np.asarray(bytearray(img_file.read()), dtype=np.uint8) img_orig = cv2.imdecode(file_bytes, 1) h, w = img_orig.shape[:2] # --- CONTROLES --- st.sidebar.subheader("ðŸŽ¯ 1. Ubicar Centro") cx = st.sidebar.slider("Centro X", 0, w, w // 2) cy = st.sidebar.slider("Centro Y", 0, h, h // 2) st.sidebar.subheader("ðŸ“ 2. Ajustar Escala") # Este punto debe coincidir con la marca de 60Â° del informe x60 = st.sidebar.slider("Alinear con Marca 60Â°", cx, w, cx + 250) # MATEMÃTICA: Equidistancia garantizada (divisiÃ³n por 6) distancia_60 = float(x60 - cx) px_cada_10 = distancia_60 / 6.0 st.sidebar.subheader("ðŸ”¬ 3. DetecciÃ³n") umbral = st.sidebar.slider("Umbral de Negros", 50, 255, 140) # --- PROCESAMIENTO --- # Creamos una copia para los colores de fondo (Sectores) img_color = img_orig.copy() gris = cv2.cvtColor(img_orig, cv2.COLOR_BGR2GRAY) _, binaria = cv2.threshold(gris, umbral, 255, cv2.THRESH_BINARY_INV) puntos_totales = 0 # 1. PINTAR SECTORES (Primero los colores para que queden de fondo) for i in range(1, 5): r_ext = int(i * px_cada_10) r_int = int((i - 1) * px_cada_10) for s in range(8): ang = s * 45 mask = np.zeros((h, w), dtype=np.uint8) cv2.ellipse(mask, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, 255, -1) cv2.circle(mask, (cx, cy), r_int, 0, -1) area_negra = np.count_nonzero(cv2.bitwise_and(binaria, mask)) area_total = np.count_nonzero(mask) if area_total > 0: ocu = (area_negra / area_total) * 100 if ocu >= 8: # Amarillo overlay = img_color.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, (0, 255, 255), -1) img_color = cv2.addWeighted(overlay, 0.4, img_color, 0.6, 0) puntos_totales += 10 elif 1.2 < ocu < 8: # Celeste overlay = img_color.copy() cv2.ellipse(overlay, (cx, cy), (r_ext, r_ext), 0, ang, ang + 45, (255, 255, 0), -1) img_color = cv2.addWeighted(overlay, 0.4, img_color, 0.6, 0) puntos_totales += 5 # 2. DIBUJAR GRILLA ROJA (Al final, encima de todo) # Definimos ROJO PURO en formato BGR para OpenCV ROJO_INTENSO = (0, 0, 255) # Anillos equidistantes (10, 20, 30, 40, 50, 60) for i in range(1, 7): radio = int(i * px_cada_10) cv2.circle(img_color, (cx, cy), radio, ROJO_INTENSO, 4) # Grosor 4 # Ejes y divisiones de sectores en ROJO for s in range(8): rad = np.radians(s * 45) x_end = int(cx + (6 * px_cada_10) * np.cos(rad)) y_end = int(cy + (6 * px_cada_10) * np.sin(rad)) cv2.line(img_color, (cx, cy), (x_end, y_end), ROJO_INTENSO, 2) # Marca de centro cv2.drawMarker(img_color, (cx, cy), (255, 255, 255), cv2.MARKER_CROSS, 30, 2) # --- RESULTADO FINAL --- st.subheader(f"ðŸ“Š Suma: {puntos_totales} pts | Incapacidad: {round((puntos_totales/320)*100, 1)}%") # IMPORTANTE: Convertimos de BGR a RGB para que Streamlit muestre el Rojo correctamente img_final = cv2.cvtColor(img_color, cv2.COLOR_BGR2RGB) st.image(img_final, use_container_width=True) st.success(f"Equidistancia: {round(px_cada_10, 2)} pÃ­xeles por cada 10 grados.") else: st.info("SubÃ­ la imagen para ver la grilla roja.")
